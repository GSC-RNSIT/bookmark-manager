$(document).ready( function(){ // Nav-pill controlls. $('.bm_collapser').click( function(){ $('.category-sub-container').hide(); }); $('.bm_expander').click( function(){ $('.category-sub-container').show(); }); // Toggle bookmark visibility. $('.progress_style_home').click( function(){ var cat_num = '#cat_num_' + $(this).attr('num'); $(cat_num).toggle(); }); $('.progress_style_edit').click( function(){ if ($(e.target).hasClass('cat_glyph_style')) return; var cat_num = '#cat_num_' + $(this).attr('num'); $(cat_num).toggle(); }); // Change style for different modes. $('.night_mode').click( function(){ $('body').css('background-color', '#1c1c1c'); $('.bm_style').css('color', '#ffb90f').find('a').css('color', '#ffb90f'); }); $('.bright_mode').click( function(){ $('body').css('background-color', 'rgba(238,173,14, 0.1)'); $('.bm_style').css('color', '').find('a').css('color', ''); }); // Override the default cursor. $('.glyphicon').hover( function(){ $(this).css('cursor', 'pointer'); }); $('.progress_style_home').hover( function(){ $(this).css('cursor', 'pointer'); }); $('.progress_style_edit').hover( function(){ $(this).css('cursor', 'pointer'); }); // Move stuff around in the edit page. $('.bm_del').dblclick( function(){ window.location.href='/b/trash_bookmark/' + this.id; }); $('.bm_up').click( function(){ window.location.href='/b/bookmark_up/' + this.id; }); $('.bm_down').click( function(){ window.location.href='/b/bookmark_down/' + this.id; }); $('.bm_random').click( function(){ window.location.href='/b/random_glyphicon/' + this.id; }); $('.cat_del').dblclick( function(){ window.location.href='/b/delete_category/' + this.id; }); });
